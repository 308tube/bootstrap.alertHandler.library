bootstrap.alertHandler.js

This is a javascript library that will allow you to build "alert" messages

To run this library you will need:

  • Bootstrap version 2.0 from Twitter
  • jQuery Project
  • bootstrap.alertHandler.js - version: 0.07.01 (2011-FEB-24)

Examples 1: basic

How to display a error message, must basic example. Bootstrap default color is yellow. Examples of how to change colors is demo later.
In the bootstrap.alertHandler.js is a Msg Code Library; you can add your own error codes into the library. For more information on that, go the Msg Code Library page. (Top Navigation bar) 

//HTML div tag with a ID of example1
<div id="example1"></div>

// must basic syntax
// "example1" is the div id
// "0" is the error message retrieved from the Msg Code Library
_BSMsg("example1").errMsg(0).build();

//to close the alert message, click the X on the right-hand side of the message.

Examples 2: something better

The method build(para) can accept the follow parameters. "slow" "fast" or a numeric number. 

//HTML div tags with a ID
<div id="example2a"></div>
<div id="example2b"></div>

//example2a
// "slow" and "fast" is valid parameter
_BSMsg("example2a").errMsg(-1).build("slow");

//example2b
//define your own time, numbers is a valid parameter, 3000 = 3 seconds
_BSMsg("example2b").errMsg(1).build(3000);

Examples 3: ad hoc messages

You do not need to add your message to the Msg Code Library. You can add the message directly into the errMsg() method.
This is not recommend for "ERROR" messages. They should be standardize and easy to find in the code and logged for analysis. 

//HTML div tags with a ID
<div id="example3a"></div>
<div id="example3b"></div>

//example3a
//define your own message
_BSMsg("example3a").errMsg({
	display_msg: "This message will be displayed to the user",
	sys_msg: "This is the geek message that will be logged"
})
.build("fast");

//example3b
_BSMsg("example3b").errMsg({
		display_msg: "How now brown cow",
		sys_msg: "var cow=brown, error cow is not brown"
	})
	.build(1000);

Examples 4: choose your color!

Choose from 4 colors that Bootstrap supports.

  • blue, green, yellow or red
  • The order you choose to call the method does not matter. As long as its before the build() method. 
//HTML div tags with a ID
<div id="example4a"></div>
<div id="example4b"></div>
<div id="example4c"></div>
<div id="example4d"></div>

//example4a
_BSMsg("example4a").errMsg(0).setColor("blue").build("slow");

//example4b
_BSMsg("example4b").setColor("red").errMsg(-1).build("fast");

//example4c
_BSMsg("example4c").errMsg(1).setColor("green").build("slow");

//example4d

		display_msg: "How now brown cow",
		sys_msg: "var cow=brown, error cow is not brown"
	})
	.setColor("yellow")
	.build("fast");

Examples 5: set display time

Choose you can choose how long the message is displayed

  • setDisplayTime(1000) method accept a number parameter. 1000 = 1 second
  • setDisplayTime(0) a parameter of zero means the message will stay on the screen until the users closes it.
  • The order you choose to call the method does not matter. As long as its before the build() method. 
//HTML div tags with a ID
<div id="example5a"></div>
<div id="example5b"></div>
<div id="example5c"></div>
<div id="example5d"></div>

//example5a
_BSMsg("example5a").setDisplayTime(0).errMsg(0).setColor("blue").build("fast");
_BSMsg("example5b").setDisplayTime(3000).setColor("red").errMsg(-1).build("fast");
_BSMsg("example5c").setDisplayTime(2000).errMsg(1).setColor("green").build("fast");
_BSMsg("example5d").setDisplayTime(1000)
	.errMsg({
		display_msg: "How now brown cow",
		sys_msg: "var cow=brown, error cow is not brown"
	})
	.setColor("yellow")
	.build("fast");

Examples 6: chaining multiple messages

Using the setElementID() method, you can chain to other message div tags.

  • setElementID(id) method accept a <div id="id"> ID. No # sign is needed.
  • When using the setElementID() method, the order in relation to the build() method matters. 
//HTML div tags with a ID
<div id="example6a"></div>
<div id="example6b"></div>
<div id="example6c"></div>
<div id="example6d"></div>

//example6a
//instead of calling 4 separate _BSMsg functions, we used one to create 4 messages.
_BSMsg("example6a").setDisplayTime(4000).errMsg(0).setColor("blue").build("fast")
	.setElementID("example6b").setDisplayTime(3000).setColor("red").errMsg(-1).build("fast")
	.setElementID("example6c").setDisplayTime(2000).errMsg(1).setColor("green").build("fast")
	.setElementID("example6d").setDisplayTime(1000)
		.errMsg({
				display_msg: "How now brown cow",
				sys_msg: "var cow=brown, error cow is not brown"
		})
		.setColor("yellow")
		.build("fast");

Advance 7: link BSMsg to a variable

Note: this should only be done if you are going to build many messages in a short period of time. In this example will we hold the _BSMsg in memory by linking it with a variable.
So using a variable as a "pointer" to our object.

In the example below, pretend your code is checking through the status of 12 positions and you need to report back what the status is. 

//code here to check on pos1 status
var status_check = _BSMsg("pos1").setDisplayTime(5000).errMsg(2).setColor("green").build("fast");

//code here to check on pos2 status
status_check.setElementID("pos2").build("fast");

//code here to check on pos3 status
status_check.setElementID("pos3").errMsg(3).setDisplayTime(0).setColor("yellow").build("fast");

//more code, error found
status_check.setElementID("pos4").errMsg(-2).setColor("red").build("fast");

//more code, system ready
status_check.setElementID("pos5").setDisplayTime(5000).errMsg(2).setColor("green").build("fast");

//more code, system ready
status_check.setElementID("pos6").build("fast"); //reuses values

//more code, system ready
status_check.setElementID("pos7").build("fast"); //reuses values

//more code, system ready
status_check.setElementID("pos8").build("fast"); //reuses values

//more code, found error
status_check.setElementID("pos9").errMsg(-2).setColor("red").setDisplayTime(0).build("fast");

//more code, system not ready
status_check.setElementID("pos10").errMsg(3).setColor("yellow").build("fast");

//more code, found error
status_check.setElementID("pos11").errMsg(-2).setColor("red").build("fast");

//more code, system ready
status_check.setElementID("pos12").errMsg(2).setColor("green").setDisplayTime(5000).build("fast");

Advance 8: another example of variable linking

Note: this should only be done if you are going to build many messages in a short period of time. In this example will we hold the _BSMsg in memory by linking it with a variable.
So using a variable as a "pointer" to our object.

In the example below, pretend your code is checking through the status of 12 positions and you need to report back what the status is. 

//build GREEN status and hold in memory by tieing it to a variable
var green_status = _BSMsg("pos1b").setDisplayTime(5000).errMsg(2).setColor("green").build("fast");

green_status.setElementID("pos2b").build("fast");

//build YELLOW status and hold in memory by tieing it to a variable
var yellow_status = _BSMsg("pos3b").errMsg(3).setDisplayTime(0).setColor("yellow").build("fast");

//build RED status and hold in memory by tieing it to a variable
var red_status = _BSMsg("pos4b").errMsg(-2).setColor("red").build("fast");

green_status.setElementID("pos5b").build("fast");
green_status.setElementID("pos6b").build("fast");
green_status.setElementID("pos7b").build("fast");
green_status.setElementID("pos8b").build("fast");

red_status.setElementID("pos9b").build("fast");

yellow_status.setElementID("pos10b").build("fast");

red_status.setElementID("pos11b").build("fast");

green_status.setElementID("pos12b").build("fast");
